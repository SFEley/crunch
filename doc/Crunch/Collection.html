def document(id_or_query)

    Document.send :new, self, query: id_or_query
  else
    Document.send :new, self, id: id_or_query
  end
end

def insert(data)
  if data['_id']
    fieldset = Fieldset.new data
  else
    fieldset = Fieldset.new data.merge('_id' => Crunch.oid)
  end
  EventMachine.next_tick do
    message = InsertMessage.new(self, fieldset)
    database << message
  end
  Document.send :new, self, data: fieldset
end

- (Object) update(opts = {})

Updates multiple records in the collection with the given modifiers. Defaults ‘multi’ to true, because if you’re updating the collection you usually want to update more than one record. If you want to update just one document, call Document#update on that object instead. Returns the selector fieldset, in case you want to query on the same documents to verify results.

Parameters:

  • (optional, Hash) opts (defaults to: {}) —

    Attribute parameters

Options Hash (opts):

  • (Fieldset, Hash) :selector — default: {} —

    Describes the document(s) to be updated (if not given, all documents in the collection will be updated)

  • (Object) :id —

    If specified, the message’s selector will include => val (not recommended at the Collection level)

  • (Fieldset, Hash) :update — default: {} —

    The values we’re updating — when given at the Collection level, use a hash of atomic update operators (i.e. ’$set’ and friends)

  • (Boolean) :upsert —

    If true, will create a new record if no document matches the selector (a blank selector matches the first document)

  • (Boolean) :multi — default: true —

    If true, updates ALL documents matching the selector rather than the first

def update(opts={})
  update = {}
  if opts[:id]
    update[:selector] = Fieldset.new (opts[:selector] || {}).merge('_id' => opts[:id])
  else
    update[:selector] = Fieldset.new opts[:selector]
  end
  update[:update] = opts[:update].is_a?(Fieldset) ? opts[:update] : Fieldset.new(opts[:update])
  update[:upsert] = opts[:upsert] if opts.has_key?(:upsert)
  update[:multi] = opts.has_key?(:multi) ? opts[:multi] : true
  message = UpdateMessage.new(self, update)
  EventMachine.next_tick do
    database << message
  end
end
